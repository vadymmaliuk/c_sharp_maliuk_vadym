using System;

class Program
 {
    static void Main(string[] args)
    {
        int x, y, z;
        Console.WriteLine("Введiть будь якi 3 цифри");
        Console.WriteLine("x = ");
        x = int.Parse(Console.ReadLine());
        Console.WriteLine("y = ");
        y = int.Parse(Console.ReadLine());
        Console.WriteLine("z = ");
        z = int.Parse(Console.ReadLine());
        if(x + y == z)
        {
            Console.WriteLine("Цифри утворюють арефметичну прогресiю");
        }
        else
        {
            Console.WriteLine("Цифри не утворюють арефметичну прогресiю");
        }
    }
}
